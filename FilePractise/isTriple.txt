public static int isTriple(int[] a)
        {
            for (int i = 0; i < a.Length; i++)
            {
                int count = 0;
                for (int j = 0; j < a.Length; j++)
                {
                    if (a[i] == a[j])
                    {
                        count++;
                    }
                }
                if (count != 3)
                {
                    return 0;
                }
            }
            return 1;
        }
